The MCP tool framework has been enhanced with stateful capabilities using
Server-Sent Events (SSE) and server-side push. This allows tools to maintain
session state and push real-time updates to clients.

🔌 NEW API ENDPOINTS
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

Session Management:
  POST   /api/tool/{tool_name}/session/create
  DELETE /api/tool/{tool_name}/session/{session_id}
  GET    /api/tool/{tool_name}/session/{session_id}/state
  PUT    /api/tool/{tool_name}/session/{session_id}/state

Real-time Updates:
  GET    /api/tool/{tool_name}/session/{session_id}/sse    (SSE Stream)
  POST   /api/tool/{tool_name}/session/{session_id}/push   (Server Push)


📊 CONFIGURATION OPTIONS
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

Add to tool's JSON configuration:

{
  "stateful_enabled": true,     // Enable stateful sessions
  "sse_enabled": true,           // Enable Server-Sent Events
  "push_enabled": true,          // Enable server-side push
  "session_timeout": 3600,       // Session timeout (seconds)
  "max_sessions": 100            // Max concurrent sessions
}

🎯 USAGE EXAMPLE
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

Python Client:
┌────────────────────────────────────────────────────────────────────────────┐
│  # Create session                                                          │
│  session = client.create_session('bank_of_canada_tool',                   │
│      initial_state={'currency': 'USD', 'threshold': 1.30})                 │
│                                                                            │
│  # Subscribe to updates                                                    │
│  client.call_tool('bank_of_canada_tool', 'subscribe_rate_updates',        │
│      {'session_id': session_id})                                           │
│                                                                            │
│  # Listen for SSE events                                                   │
│  client.listen_sse('bank_of_canada_tool', {                               │
│      'update': lambda d: print(f"Rate: {d}"),                              │
│      'alert': lambda d: print(f"ALERT: {d}")                               │
│  })                                                                        │
│                                                                            │
│  # Close when done                                                         │
│  client.close_session('bank_of_canada_tool')                              │
